# Install Node JS
*Verify Node version\
node --version\

echo $path\

# Install Angular Cli
*For linux use - sudo npm i -g @angular/cli \ 
For Windows use - npm i -g @angular/cli \
(i refers install | g refers install globally)\
Verify Angular Cli version\
ng --version*
